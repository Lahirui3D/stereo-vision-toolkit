49 bool StereoCameraFromVideo::captureSingle(){
50  int current_frame = stream.get(CV_CAP_PROP_POS_FRAMES);
51  if (current_frame >= number_frames ){
52  setPosition(0);
53  }
54 
55  bool res = stream.read(image_buffer);
56 
57  if (res) {
58  // Simulate frame rate
59  double delay_needed = (1000.0/(video_fps+1)) - frame_timer.elapsed();
60  if(delay_needed > 0){
61  QThread::msleep(delay_needed);
62  }
63  emit videoPosition(100*((float) stream.get(CV_CAP_PROP_POS_FRAMES))/number_frames);
64  if(image_buffer.channels() == 3)
65  cv::cvtColor(image_buffer, image_buffer, CV_RGB2GRAY);
66 
67  cv::Mat(image_buffer,
68  cv::Rect(0, 0, image_buffer.cols / 2, image_buffer.rows))
69  .copyTo(left_raw);
70  cv::Mat(image_buffer, cv::Rect(image_buffer.cols / 2, 0,
71  image_buffer.cols / 2, image_buffer.rows))
72  .copyTo(right_raw);
73 
74  frame_timer.restart();
75  }
76  if (!res){
77  send_error(CAPTURE_ERROR);
78  emit captured_fail();
79  } else {
80  emit captured_success();
81  }
82  emit captured();
83  return res;
84 }
