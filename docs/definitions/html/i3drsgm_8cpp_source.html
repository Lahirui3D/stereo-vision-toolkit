406  }
407  pyramid_num = params.nNumberOfPyramids;
408  }
409 
410  top_pyramid = params.nNumberOfPyramids - pyramid_num;
411 
412  int j = params.nNumberOfPyramids - 1;
413  for (int i = 0; i <= params.nNumberOfPyramids; i++)
414  {
415  if (i < pyramid_num)
416  {
417  EditPyramidParamRaw(&this->params_raw, j, param_name, param_val_true);
418  //std::cerr << "p: " << j << std::endl;
419  }
420  else
421  {
422  EditPyramidParamRaw(&this->params_raw, j, param_name, param_val_false);
423  }
424  //std::cerr << "i: " << i << " j: " << j << std::endl;
425  j--;
426  }
427 
428  if (include_subpix)
429  {
430  enableSubpixel(true);
431  }
432  else
433  {
434  enableSubpixel(false);
435  }
436 
437  this->matcher_status = createMatcher();
