20  CLSCTX_INPROC_SERVER, IID_PPV_ARGS(&pDevEnum));

39 
40  VARIANT var;

42

45  if (FAILED(hr)) {
46  hr = pPropBag->Read(L"FriendlyName", &var, 0);
47  }
48 
49  if (SUCCEEDED(hr)) {
50  std::wstring str(var.bstrVal);
51  if (std::string(str.begin(), str.end()) == friendly_name) {
52  std::string device_path_serial = get_device_path_serial(pMoniker);
53  bool device_known = false;
54  AbstractStereoCamera::StereoCameraSerialInfo serial_info;
55  for (auto& known_serial_info : known_serial_infos) {
56  if (device_path_serial == known_serial_info.left_camera_serial){
57  serial_info = known_serial_info;
58  device_known = true;
59  break;
60  }
61  }
62  if (device_known){
63  connected_serial_infos.push_back(serial_info);
64  }
65  i++;
66  }
67  VariantClear(&var);
68  }
69  hr = pPropBag->Write(L"FriendlyName", &var);
70 
71  pPropBag->Release();

73  }
