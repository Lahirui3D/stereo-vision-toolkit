21 cv::Scalar CameraDisplayWidget::hsv2bgr(cv::Scalar hsv){
22  /*
23  H(Hue): 0-360 degrees
24  S(Saturation): 0-100 percent
25  V(Value): 0-100 percent
26 
27  R(Red): 0-255
28  G(Green): 0-255
29  B(Blue): 0-255
30  */
31  float H = hsv[0];
32  float S = hsv[1];
33  float V = hsv[2];
34  if(H>360 || H<0 || S>100 || S<0 || V>100 || V<0){
35  return cv::Scalar(0,0,0);
36  }
37  float s = S/100;
38  float v = V/100;
39  float C = s*v;
40  float X = C*(1-abs(fmod(H/60.0, 2)-1));
41  float m = v-C;
42  float r,g,b;
43  if(H >= 0 && H < 60){
44  r = C,g = X,b = 0;
45  }
46  else if(H >= 60 && H < 120){
47  r = X,g = C,b = 0;
48  }
49  else if(H >= 120 && H < 180){
50  r = 0,g = C,b = X;
51  }
52  else if(H >= 180 && H < 240){
53  r = 0,g = X,b = C;
54  }
55  else if(H >= 240 && H < 300){
56  r = X,g = 0,b = C;
57  }
58  else{
59  r = C,g = 0,b = X;
60  }
61  int R = (r+m)*255;
62  int G = (g+m)*255;
63  int B = (b+m)*255;
64  return cv::Scalar(B,G,R);
65 }

73 
74  cv::Mat display_image = new_image.clone();
75  if (this->show_epipolar_lines){
76  // draw epipilar lines on image
77  int lineType = cv::LINE_8;

79  int lineSpacing = display_image.size().height/numOfLines;
80  int thickness = (int)lineSpacing/100.0;
81  if (thickness <= 1){ thickness = 1;}
82  for (int i = 0; i < numOfLines-1; i++){
83  int line_y = lineSpacing * (i+1);
84  cv::Point start(0,line_y);
85  cv::Point end(display_image.size().width,line_y);
86  int hue = (int)(360.0/numOfLines)*i;
87  cv::Scalar hsv_color(hue, 100, 100);
88  cv::Scalar bgr_color = hsv2bgr(hsv_color);
89  cv::line( display_image,
90  start,
91  end,
92  bgr_color,

95  }
