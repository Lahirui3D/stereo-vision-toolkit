15 void CStereoCameraBaslerImageEventHandler::OnImageGrabbed( Pylon::CInstantCamera& camera, const Pylon::CGrabResultPtr& ptrGrabResult)
16 {
17  const auto now = std::chrono::system_clock::now();
18  int frame_timestamp_milli = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
19  // left camera
20  if (camera.GetDeviceInfo().GetSerialNumber() == this->left_serial.c_str()){
21  bool res = PylonSupport::grabImage2mat(ptrGrabResult,this->formatConverter,this->live_left_image);
22  if (res){
23  timestamp_left = frame_timestamp_milli;
24  }
25  }
26  // right camera
27  if (camera.GetDeviceInfo().GetSerialNumber() == this->right_serial.c_str()){
28  bool res = PylonSupport::grabImage2mat(ptrGrabResult,this->formatConverter,this->live_right_image);
29  if (res){
30  timestamp_right = frame_timestamp_milli;
31  }
32  }
33  mtx.lock();
34  int lr_diff = abs(timestamp_left - timestamp_right);
35  if (lr_diff <= 10){
36  newFrames = true;
37  this->left_image = this->live_left_image.clone();
38  this->right_image = this->live_right_image.clone();
39  } else {
40  newFrames = false;
41  }
42  mtx.unlock();
43 }

45 bool CStereoCameraBaslerImageEventHandler::getImagePair(cv::Mat &left, cv::Mat &right, int timeout=1000){
46  int time_count = 0;
47  bool timed_out = true;
48  while(time_count < timeout){
49  mtx.lock();
50  if (newFrames){
51  left = this->left_image.clone();
52  right = this->right_image.clone();
53  newFrames = false;
54  timed_out = false;
55  mtx.unlock();
56  break;
57  }
58  mtx.unlock();
59  std::this_thread::sleep_for(std::chrono::milliseconds(1));
60  time_count += 1;
61  }
62  return !timed_out;
63 }

103  autoGain = true;
104  } else {
105  autoGain = false;

107 
108  try
109  {
110 
111  // Create an instant camera object with the camera device found first.

119 
120  std::string camera_left_serial = stereoCameraSerialInfo_.left_camera_serial;
121  std::string camera_right_serial = stereoCameraSerialInfo_.right_camera_serial;
122 
123  bool cameraLeftFind = false;
124  for (size_t i = 0; i < all_cameras.GetSize(); ++i)
125  {
126  all_cameras[i].Attach(tlFactory.CreateDevice(devices[i]));

138  return false;
139  }
140 
141  bool cameraRightFind = false;
142  for (size_t i = 0; i < all_cameras.GetSize(); ++i)
143  {
144  all_cameras[i].Attach(tlFactory.CreateDevice(devices[i]));
145 
146  if (all_cameras[i].GetDeviceInfo().GetSerialNumber() == camera_right_serial.c_str())
147  {
148  cameras->operator[](1).Attach(tlFactory.CreateDevice(devices[i]));
149  cameraRightFind = true;
150  break;
151  }
152  }
153 
154  if (!cameraRightFind){
155  std::cerr << "Failed to find right camera with serial: " << camera_right_serial << std::endl;
156  return false;
157  }
158 
159  // Set device link throughput to fix data collision
160  enableDeviceLinkThroughputLimit(true);
161  setDeviceLinkThroughput(100000000);
162 
163  //check open
164  bool connected = true;
165  try{

168  cameras->operator[](i).Open();
169  connected &= cameras->operator[](i).IsOpen();

171  }
172  catch (const Pylon::GenericException &e)

174  // Error handling.

177  connected = false;
178  }
179 
180  if (!connected){
181  return connected;
182  }
183 
184  getImageSize(cameras->operator[](0),image_width,image_height,image_bitdepth);
185  emit update_size(image_width, image_height, image_bitdepth);
186 
187  formatConverter = new Pylon::CImageFormatConverter();
188 
189  //formatConverter->OutputPixelFormat = Pylon::PixelType_Mono8;
190  formatConverter->OutputPixelFormat = Pylon::PixelType_BGR8packed;
191  formatConverter->OutputBitAlignment = Pylon::OutputBitAlignment_MsbAligned;
192 
193  imageEventHandler = new CStereoCameraBaslerImageEventHandler(camera_left_serial,camera_right_serial,formatConverter);
194 
195  for (size_t i = 0; i < cameras->GetSize(); ++i)
196  {
197  //cameras->operator[](i).MaxNumBuffer = 1;
198  //cameras->operator[](i).OutputQueueSize = 1;
199  //cameras->operator[](i).ClearBufferModeEnable();
200  cameras->operator[](i).RegisterImageEventHandler( imageEventHandler, Pylon::RegistrationMode_Append, Pylon::Cleanup_Delete);
201  }
202  }
203  catch (const Pylon::GenericException &e)
204  {
205  // Error handling.
206  std::cerr << "An exception occurred whilst setting up cameras." << std::endl
207  << e.GetDescription() << std::endl;
208  connected = false;
209  return connected;
210  }
211 
212  // Set device parameters
213  setBinning(binning);
214  enableTrigger(trigger);
215  setPacketSize(packet_size);
216  setFPS(fps);
217 
218  enableAutoGain(autoGain);
219  enableAutoExposure(autoExpose);
220  setExposure(exposure);
221  setGain(gain);
222  setPacketDelay(packet_delay);
223 
224  //lineStatusTimerId = startTimer(lineStatusTimerDelay);
225 
226  connected = true;
227  return connected;
