87 void CalibrateFromImagesDialog::setLeftImages() {
88  QModelIndex parentIndex = left_file_model->index(left_path);
89  int numRows = left_file_model->rowCount(parentIndex);
90 
91  for (int row = 0; row < numRows; ++row) {
92  QModelIndex childIndex = left_file_model->index(row, 0, parentIndex);
93  QString path = left_file_model->data(childIndex).toString();
94  QString fname = left_path + "/" + path;
95  cv::Mat im = cv::imread(fname.toStdString(), cv::IMREAD_GRAYSCALE);
96  //left_image_list.append(left_path + "/" + path);
97  left_image_list.append(im);
98  }
99 
100  checkImageCount();
101 }

119 void CalibrateFromImagesDialog::findImages() {
120  if (left_path != "") {
121  QStringList leftFilters;
122 
123  if (left_mask == "") {
124  leftFilters << "*.png"
125  << "*.tif"
126  << "*.jpg"
127  << "*.jpeg";
128  } else {
129  leftFilters << left_mask;
130  }
131 
132  left_file_model->setNameFilters(leftFilters);
133  left_file_model->setRootPath(left_path);
134  left_file_model->setNameFilterDisables(false);
135 
136  ui->leftImageTable->setModel(left_file_model);
137  ui->leftImageTable->setRootIndex(left_file_model->index(left_path));
138 
139  } else {
140  ui->leftImageTable->clearSpans();
141  }
142 
143  if (right_path != "") {
144  QStringList rightFilters;
145 
146  if (right_mask == "") {
147  rightFilters << "*.png"
148  << "*.tif"
149  << "*.jpg"
150  << "*.jpeg";
151  } else {
152  rightFilters << right_mask;
153  }
154 
155  right_file_model->setNameFilters(rightFilters);
156  right_file_model->setRootPath(right_path);
157  right_file_model->setNameFilterDisables(false);
158 
159  ui->rightImageTable->setModel(right_file_model);
160  ui->rightImageTable->setRootIndex(right_file_model->index(right_path));
161  } else {
162  ui->rightImageTable->clearSpans();
163  }
164 }

166 void CalibrateFromImagesDialog::checkImageCount(void){
167  // Only allow user to continue if we've got enough cal images
168  int n_left_images = left_image_list.size();
169  int n_right_images = right_image_list.size();
170 
171  if(n_left_images >= 1 && n_right_images >= 1 && n_left_images == n_right_images){
172  ui->okPushButton->setEnabled(true);

174  ui->okPushButton->setEnabled(false);
