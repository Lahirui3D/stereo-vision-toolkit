159 void AbstractStereoCamera::reproject3D() {
160 
161  cv::Mat disparity_downscale;
162  cv::Mat disparity;
163 
164  matcher->getDisparity(disparity);
165  disparity.copyTo(disparity_downscale);
166 
167  cv::Mat depth;
168  matcher->calcDepth(disparity,depth);
169 
170  qDebug() << "disparity image size: " << disparity_downscale.size().width << "," << disparity_downscale.size().height;
171 
172  //cv::abs(disparity_downscale);
173 
174  // By default the disparity maps are scaled by a factor of 16.0
175  disparity_downscale /= 16.0;
176 
177  int row = (disparity_downscale.rows-1)/2;
178  int column = (disparity_downscale.cols-1)/2;
179  float disp_val_f = disparity_downscale.at<float>(row, column);
180  qDebug() << "IMAGE CENTER: " << column+1 << "," << row+1;
181  qDebug() << "DISPARITY AT CENTER: " << disp_val_f;
182 
183  if (Q.empty() || disparity_downscale.empty()) {
184  qDebug() << "Q or disparity map is empty";
185  return;
186  }
187 
188  //cv::reprojectImageTo3D(disparity_downscale, stereo_reprojected, Q, true);
189 
190  //qDebug() << "reprojected image size: " << stereo_reprojected.size().width << "," << stereo_reprojected.size().height;
191 
192  pcl::PointCloud<pcl::PointXYZRGB>::Ptr ptCloudTemp(
193  new pcl::PointCloud<pcl::PointXYZRGB>);
194 
195  pcl::PointXYZRGB point;
196  uint32_t rgb = 0;
197 
198  point.x = 0;
199  point.y = 0;
200  point.z = 0;
201 
202  rgb = ((int)255) << 16 | ((int)255) << 8 | ((int)255);
203  point.rgb = *reinterpret_cast<float *>(&rgb);
204  ptCloudTemp->points.push_back(point);
205 
206  cv::Matx44d _Q;
207  Q.convertTo(_Q, CV_64F);
208 
209  for (int i = 0; i < disparity_downscale.rows; i++)
210  {
211  for (int j = 0; j < disparity_downscale.cols; j++)
212  {
213  float d = disparity_downscale.at<float>(i, j);
214 
215  if (d < 10000 && d > 0){
216  float x_index = j;
217  float y_index = i;
218 
219  //qDebug() << d;
220 
221  cv::Vec4d homg_pt = _Q * cv::Vec4d((double)x_index, (double)y_index, (double)d, 1.0);
222 
223  float x = (float)homg_pt[0] / (float)homg_pt[3];
224  float y = (float)homg_pt[1] / (float)homg_pt[3];
225  float z = (float)homg_pt[2] / (float)homg_pt[3];
226 
227  uchar intensity = left_remapped.at<uchar>(i,j);
228 
229  pcl::PointXYZRGB point;
230  point.x = x;
231  point.y = y;
232  point.z = z;
233  point.r = intensity;
234  point.g = intensity;
235  point.b = intensity;
236  ptCloudTemp->points.push_back(point);
237  }
238  }
239  }
240 
241  if(ptCloudTemp->points.empty()){
242  qDebug() << "Failed to create Point cloud. Point cloud is empty";
243  return;
244  }
245 
246  pcl::PassThrough<pcl::PointXYZRGB> pass;
247  pass.setInputCloud (ptCloudTemp);
248  pass.setFilterFieldName ("z");
249  pass.setFilterLimits(visualisation_min_z, visualisation_max_z);
250  pass.filter(*ptCloudTemp);
251 
252  /*
253  pcl::RadiusOutlierRemoval<pcl::PointXYZRGB> outrem;
254  outrem.setInputCloud(ptCloudTemp);
255  outrem.setRadiusSearch(0.05);
256  outrem.setMinNeighborsInRadius (5);
257  outrem.filter (*ptCloudTemp);
258 
259  pcl::StatisticalOutlierRemoval<pcl::PointXYZRGB> sor;
260  sor.setInputCloud (ptCloudTemp);
261  sor.setMeanK (5);
262  sor.setStddevMulThresh (4.0);
263  sor.filter(*ptCloudTemp);
264  */
265 
266  ptCloud = ptCloudTemp;
267 
268  qDebug() << "tmp point cloud size: " << ptCloudTemp->points.size();
269  qDebug() << "point cloud size: " << ptCloud->points.size();
270 
271  emit reprojected();
272 }

488 bool AbstractStereoCamera::loadCalibrationYamlFiles(QString left_cal, QString right_cal) {
489  // implimented from (https://github.com/i3drobotics/pyStereo3D/blob/master/Stereo3D/Stereo3D/StereoCalibration.py)
490  int flags = cv::FileStorage::READ;
491 
492  cv::FileStorage fs_l(left_cal.toStdString(), flags);
493 
494  if (fs_l.isOpened()) {
495  fs_l["image_width"] >> cal_image_width;
496  fs_l["image_height"] >> cal_image_height;
497  fs_l["camera_matrix"] >> l_camera_matrix;
498  fs_l["distortion_coefficients"] >> l_dist_coeffs;
499  fs_l["rectification_matrix"] >> l_rect_mat;
500  fs_l["projection_matrix"] >> l_proj_mat;
501  } else {
502  return false;
503  }
504 
505  fs_l.release();
506 
507  if (cal_image_width != image_width || cal_image_height != image_height){
508  qDebug() << "Image size does not match calibrated image size";
509  return false;
510  }
511 
512  cv::FileStorage fs_r(right_cal.toStdString(), flags);
513 
514  int cal_r_image_width, cal_r_image_height;
515  if (fs_r.isOpened()) {
516  fs_r["image_width"] >> cal_r_image_width;
517  fs_r["image_height"] >> cal_r_image_height;
518  fs_r["camera_matrix"] >> r_camera_matrix;
519  fs_r["distortion_coefficients"] >> r_dist_coeffs;
520  fs_r["rectification_matrix"] >> r_rect_mat;
521  fs_r["projection_matrix"] >> r_proj_mat;
522  } else {
523  return false;
524  }
525 
526  fs_r.release();
527 
528  if (cal_r_image_width != image_width || cal_r_image_height != image_height){
529  qDebug() << "Image size does not match calibrated image size";
530  return false;
531  }
532 
533  //calculate q
534  Q = cv::Mat::zeros(4, 4, CV_64F);
535  double cx = l_proj_mat.at<double>(0,2);
536  double cxr = r_proj_mat.at<double>(0,2);
537  double cy = l_proj_mat.at<double>(1,2);
538  double fx = l_camera_matrix.at<double>(0,0);
539  double fy = l_camera_matrix.at<double>(1,1);
540 
541  double p14 = r_proj_mat.at<double>(0,3);
542  baseline = -p14 / fx;
543 
544  double q33 = -(cx - cxr) / baseline;
545 
546  Q.at<double>(0,0) = 1.0;
547  Q.at<double>(0,3) = cx;
548  Q.at<double>(1,1) = 1.0;
549  Q.at<double>(1,3) = -cy;
550  Q.at<double>(2,3) = fx;
551  Q.at<double>(3,2) = 1.0 / baseline;
552  Q.at<double>(3,3) = q33;
553 
554  Q.convertTo(Q, CV_32F);
555 
556  generateRectificationMaps(cv::Size(image_width,image_height));
557 
558  return true;
559 }
