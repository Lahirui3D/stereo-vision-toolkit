19 /*
20  Endianness
21  ---
22 
23  Although not totally clear from the docs, some of QImage's formats we use here are
24  endian-dependent. For example:
25 
26  Little Endian
27  QImage::Format_ARGB32 the bytes are ordered: B G R A
28  QImage::Format_RGB32 the bytes are ordered: B G R (255)
29  QImage::Format_RGB888 the bytes are ordered: R G B
30 
31  Big Endian
32  QImage::Format_ARGB32 the bytes are ordered: A R G B
33  QImage::Format_RGB32 the bytes are ordered: (255) R G B
34  QImage::Format_RGB888 the bytes are ordered: R G B
35 
36  Notice that Format_RGB888 is the same regardless of endianness. Since OpenCV
37  expects (B G R) we need to swap the channels for this format.
38 
39  This is why some conversions here swap red and blue and others do not.
40 
41  This code assumes little endian. It would be possible to add conversions for
42  big endian machines though. If you are using such a machine, please feel free
43  to submit a pull request on the GitHub page.
44 */
45 #if Q_BYTE_ORDER == Q_BIG_ENDIAN
46 #error Some of QImage's formats are endian-dependant. This file assumes little endian. See comment at top of header.
47 #endif

58  QImage image( inMat.data,

60  static_cast<int>(inMat.step),
61  QImage::Format_ARGB32 );

64  }
65 
66  // 8-bit, 3 channel
67  case CV_8UC3:
68  {
69  QImage image( inMat.data,
70  inMat.cols, inMat.rows,
71  static_cast<int>(inMat.step),
72  QImage::Format_RGB888 );
73 
74  return image.rgbSwapped();
75  }
76 
77  // 8-bit, 1 channel
78  case CV_8UC1:
79  {
80  #if QT_VERSION >= QT_VERSION_CHECK(5, 5, 0)
81  QImage image( inMat.data,
82  inMat.cols, inMat.rows,
83  static_cast<int>(inMat.step),
84  QImage::Format_Grayscale8 );
85  #else
86  static QVector<QRgb> sColorTable;
87 
88  // only create our color table the first time
89  if ( sColorTable.isEmpty() )
90  {
91  sColorTable.resize( 256 );
92 
93  for ( int i = 0; i < 256; ++i )
94  {
95  sColorTable[i] = qRgb( i, i, i );
96  }
97  }

100  inMat.cols, inMat.rows,
101  static_cast<int>(inMat.step),
102  QImage::Format_Indexed8 );

105  #endif
106 
107  return image;
108  }
109

127  inline cv::Mat QImageToCvMat( const QImage &inImage, bool inCloneImageData = true )
128  {
129  switch ( inImage.format() )
130  {
131  // 8-bit, 4 channel
132  case QImage::Format_ARGB32:
133  case QImage::Format_ARGB32_Premultiplied:
134  {
135  cv::Mat mat( inImage.height(), inImage.width(),
136  CV_8UC4,
137  const_cast<uchar*>(inImage.bits()),
138  static_cast<size_t>(inImage.bytesPerLine())
139  );
140 
141  return (inCloneImageData ? mat.clone() : mat);
142  }
143 
144  // 8-bit, 3 channel
145  case QImage::Format_RGB32:
146  {
147  if ( !inCloneImageData )
148  {
149  qWarning() << "ASM::QImageToCvMat() - Conversion requires cloning so we don't modify the original QImage data";
150  }
151 
152  cv::Mat mat( inImage.height(), inImage.width(),
153  CV_8UC4,
154  const_cast<uchar*>(inImage.bits()),
155  static_cast<size_t>(inImage.bytesPerLine())
156  );
157 
158  cv::Mat matNoAlpha;
159 
160  cv::cvtColor( mat, matNoAlpha, cv::COLOR_BGRA2BGR ); // drop the all-white alpha channel
161 
162  return matNoAlpha;
163  }
164 
165  // 8-bit, 3 channel
166  case QImage::Format_RGB888:
167  {
168  if ( !inCloneImageData )
169  {
170  qWarning() << "ASM::QImageToCvMat() - Conversion requires cloning so we don't modify the original QImage data";
171  }
172 
173  QImage swapped = inImage.rgbSwapped();
174 
175  return cv::Mat( swapped.height(), swapped.width(),
176  CV_8UC3,
177  const_cast<uchar*>(swapped.bits()),
178  static_cast<size_t>(swapped.bytesPerLine())
179  ).clone();
180  }
181 
182  // 8-bit, 1 channel
183  case QImage::Format_Indexed8:
184  {
185  cv::Mat mat( inImage.height(), inImage.width(),
186  CV_8UC1,
187  const_cast<uchar*>(inImage.bits()),
188  static_cast<size_t>(inImage.bytesPerLine())
189  );
190 
191  return (inCloneImageData ? mat.clone() : mat);
192  }
193 
194  default:
195  qWarning() << "ASM::QImageToCvMat() - QImage format not handled in switch:" << inImage.format();
196  break;
197  }
198 
199  return cv::Mat();
200  }
